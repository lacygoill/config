Msg: Prefer explicit `-n` to check for output (or run command without `test` to check for success).

# Bad:

    if test "$(cmd)"; then
        # `cmd` has some output
    fi

# Good:

            vv
    if test -n "$(cmd)"; then
        # `cmd` has some output
    fi

# Bad:

    if test ! "$(pidof -s vim)"; then
        # Vim is not running
    fi

# Good:

    if ! pidof -s vim; then
        # Vim is not running
    fi

# Rationale: <https://github.com/koalaman/shellcheck/wiki/SC2243>

# Regex:

    \<test\s\+\%(!\s*\)\="\=$(
