Msg: Do not append a trailing slash in a `ls(1)` command.

# Bad:

    ls /path/to/dir/
                   ^

# Good:

    ls /path/to/dir
                  ^

# Rationale: The trailing slash can prevent a symlink to be listed.

MRE:

    $ mkdir dir
    $ ln --symbolic dir symlink

                v
    $ ls symlink/
    # no output

    $ ls symlink
    symlink@

                    v
    $ ls -ld symlink/
    drwxrwxr-x ... symlink//

    $ ls -ld symlink
    lrwxrwxrwx ... symlink -> dir/

# Regex:

    \<ls\s.*\S/$
