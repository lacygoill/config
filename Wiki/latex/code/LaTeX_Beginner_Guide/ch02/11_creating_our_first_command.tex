\documentclass{article}
%            ┌ we define the macro `\TUG`:{{{
%            │
%            │         whenever  we write  `\TUG` in  the document,  LaTeX will
%            │         replace it  with the text  passed as an argument  in the
%            │         output
%            │}}}
\newcommand{\TUG}{TeX Users Group}
\begin{document}
\section{The \TUG}
The \TUG\ is an organization for people who are interested in
\TeX\ or \LaTeX.
\end{document}
